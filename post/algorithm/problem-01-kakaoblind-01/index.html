from collections import deque

def solution(record):
  answer = []
  info = []
  queue = deque()
  user = {}
  for case in record:
    # 1
    info = case.split()
    queue.append((info[0], info[1]))
    if len(info) == 3:
      user[info[1]] = info[2]

  # 2
  while(queue):
    cmd, id = queue.popleft()
    nickname = user[id]

    if cmd == "Enter":
      answer.append(nickname + "님이 들어왔습니다.")
    elif cmd == "Leave":
      answer.append(nickname + "님이 나갔습니다.")

  return answer


풀이 방법

  1. 입력을 쪼개서 (명령어, ID) 쌍을 큐에 넣고, (ID : 닉네임)을 딕셔너리에 추가했다. 이렇게 하면 최종 Change된 닉네임까지 반영할 수 있다
  2. 큐에서 하나씩 꺼내서, ID로 닉네임을 찾고, 명령어가 Enter일 때와 Leave일 때 문자열을 생성해서 answer 배열에 추가했다
